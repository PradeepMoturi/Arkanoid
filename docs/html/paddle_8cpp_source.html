111 void Paddle::CollisionChecker()
112 {
113  QList<QGraphicsItem*> cItems = collidingItems();
114  for (int i = 0, n = cItems.size(); i < n; ++i)
115  {
116  Ball* object = dynamic_cast<Ball*>(cItems[i]);
117  if (object)
118  {
119  if(object->getCenterX()>=this->leftCornerX()&&object->getCenterX()<=this->rightCornerX()&&object->y()+(object->rect().width()/2)<y())
120  {
121  double factor = fabs(this->getCenterX() - object->getCenterX())/width();
122  factor = factor*2*0.3;
123  emit ballCollision(dynamic_cast<Ball*>(cItems[i]),false,false,factor);
124  }
125  else if(object->getCenterX()>=this->leftCornerX())
126  {
127  emit ballCollision(dynamic_cast<Ball*>(cItems[i]),false,true,0);
128  }
129  else
130  {
131  emit ballCollision(dynamic_cast<Ball*>(cItems[i]),true,false,0);
132  }
133  }
134 
135  Powerup* power = dynamic_cast<Powerup*>(cItems[i]);
136  if(power)
137  {
138  if(power->powerup_id==1)
139  {
140  emit multiballadd();
141  }
142  if(power->powerup_id == 2)
143  {
144  cnt = 1;
145  paddle_width = 150;
146  this->setRect(-25,0,paddle_width,paddle_height);
147  }
148  if(power->powerup_id == 3)
149  {
150  cnt = 1;
151  paddle_width = 50;
152  this->setRect(25,0,paddle_width,paddle_height);
153  }
154  if(power->powerup_id==4)
155  {
156  emit(sendStatus());
157  }
158  emit destroy_powerup(power);
159  }
160  }
161 }
